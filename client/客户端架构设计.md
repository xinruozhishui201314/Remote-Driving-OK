2.2 架构设计原则
text

关键设计原则：
1. 渲染路径零拷贝 — 视频帧从解码到显示不经过CPU内存拷贝
2. 控制路径最短化 — 操控指令bypass所有非必要中间层
3. UI线程不阻塞 — 所有I/O、计算、解码在独立线程/线程池
4. 状态机驱动 — 用有限状态机管理所有系统状态转换
5. 优雅降级 — 网络恶化时自动降级而非崩溃

// =====================================================
// 自适应码率控制器
// =====================================================
class AdaptiveBitrateController : public QObject {
    Q_OBJECT
public:
    struct BitrateDecision {
        uint32_t targetBitrateKbps;
        uint32_t targetFps;
        VideoResolution resolution;
        QualityLevel quality;
        std::string reason;
    };
    
    // 基于Google Congestion Control (GCC) 算法的改进版
    BitrateDecision evaluate(const NetworkMetrics& metrics) {
        // 1. 基于延迟梯度的带宽估计 (Delay-based)
        double delayGradient = calculateDelayGradient(metrics);
        uint32_t delayBasedEstimate = estimateFromDelay(delayGradient);
        
        // 2. 基于丢包率的带宽估计 (Loss-based) 
        uint32_t lossBasedEstimate = estimateFromLoss(metrics.packetLossRate);
        
        // 3. 取两者较小值作为最终估计
        uint32_t estimatedBandwidth = std::min(delayBasedEstimate, 
                                                lossBasedEstimate);
        
        // 4. 平滑处理，避免码率剧烈波动
        uint32_t smoothedBitrate = applySmoothing(estimatedBandwidth);
        
        // 5. 根据码率决定分辨率和帧率
        return makeBitrateDecision(smoothedBitrate, metrics);
    }
    
private:
    // 延迟梯度检测 (Trendline Filter)
    double calculateDelayGradient(const NetworkMetrics& metrics) {
        m_delayHistory.push_back({
            metrics.timestamp,
            metrics.oneWayDelayMs
        });
        
        if (m_delayHistory.size() < kMinSamples) return 0.0;
        
        // 使用线性回归计算趋势
        return linearRegression(m_delayHistory).slope;
    }
    
    // AIMD (Additive Increase Multiplicative Decrease)
    uint32_t applyAIMD(uint32_t current, SignalState state) {
        switch (state) {
            case SignalState::Overuse:
                return static_cast<uint32_t>(current * 0.85); // 降15%
            case SignalState::Normal:
                return current + kAdditiveIncrease; // 加性增
            case SignalState::Underuse:
                return current; // 保持
        }
    }
    
    CircularBuffer<DelayPoint, 200> m_delayHistory;
    static constexpr uint32_t kAdditiveIncrease = 100; // 100kbps/step
    static constexpr size_t kMinSamples = 20;
};

/*
 * 视频数据流路径（零拷贝）:
 * 
 * [网络] → [RTP解封装] → [解码器(硬件)] → [GPU纹理] → [渲染]
 *     ↓           ↓              ↓              ↓
 *  独立IO线程  解封装线程      解码线程      渲染线程(主线程)
 *  
 * 全程数据在GPU内存中流转，CPU仅传递句柄/指针
 */

// 视频帧的零拷贝表示
class VideoFrame {
public:
    enum class MemoryType {
        CPU_MEMORY,
        GPU_TEXTURE_GL,      // OpenGL纹理
        GPU_TEXTURE_VULKAN,  // Vulkan纹理
        DMA_BUF,             // Linux DMA-BUF
        D3D11_TEXTURE,       // Windows D3D11
        VAAPI_SURFACE,       // VA-API surface
    };
    
    struct PlaneInfo {
        void* data;           // CPU指针或GPU句柄
        uint32_t stride;
        uint32_t size;
    };
    
    MemoryType memoryType;
    uint32_t width, height;
    PixelFormat format;
    int64_t pts;              // 演示时间戳
    int64_t captureTimestamp;  // 原始采集时间戳（用于延迟计算）
    std::vector<PlaneInfo> planes;
    
    // GPU纹理信息
    union {
        GLuint glTextureId;
        VkImage vkImage;
        void* d3d11Texture;
        uintptr_t vaaSurface;
    } gpuHandle;
    
    // 引用计数（零拷贝需要）
    std::shared_ptr<FramePool> pool; // 归还到池
};

// 解码器工厂 - 自动选择最佳硬件解码器
class DecoderFactory {
public:
    static std::unique_ptr<IHardwareDecoder> create(
            VideoCodec codec, 
            DecoderPreference pref = DecoderPreference::HardwareFirst) {
        
        // 按优先级尝试
        #ifdef Q_OS_LINUX
        // 1. VAAPI (Intel/AMD)
        if (auto dec = tryCreate<VAAPIDecoder>(codec)) return dec;
        // 2. V4L2 M2M (嵌入式ARM)
        if (auto dec = tryCreate<V4L2M2MDecoder>(codec)) return dec;
        // 3. NVDEC (NVIDIA)  
        if (auto dec = tryCreate<NVDECDecoder>(codec)) return dec;
        #endif
        
        #ifdef Q_OS_WIN
        // 1. D3D11VA
        if (auto dec = tryCreate<D3D11VADecoder>(codec)) return dec;
        // 2. DXVA2
        if (auto dec = tryCreate<DXVA2Decoder>(codec)) return dec;
        #endif
        
        // Fallback: FFmpeg软解
        return std::make_unique<FFmpegSoftDecoder>(codec);
    }
};

// =====================================================
// 媒体管线 (Media Pipeline)
// =====================================================
class MediaPipeline : public QObject {
    Q_OBJECT
public:
    /*
     * 管线架构：
     * 
     * ┌──────────┐   ┌──────────┐   ┌──────────┐   ┌──────────┐
     * │  Source   │──▶│Demuxer/  │──▶│ Decoder  │──▶│ Renderer │
     * │(Network) │   │DePacket  │   │(HW Accel)│   │(GPU)     │
     * └──────────┘   └──────────┘   └──────────┘   └──────────┘
     *   IO Thread     Parse Thread  Decode Thread  Render Thread
     *                                    │
     *                              ┌─────▼──────┐
     *                              │ Frame Pool │ (预分配GPU buffer)
     *                              └────────────┘
     */
    
    bool initialize(const PipelineConfig& config) {
        // 1. 创建帧池（预分配GPU buffer，避免运行时分配）
        m_framePool = std::make_shared<FramePool>(
            config.framePoolSize, // 通常8-16帧
            config.maxWidth, config.maxHeight,
            config.gpuMemoryType
        );
        
        // 2. 创建硬件解码器
        m_decoder = DecoderFactory::create(config.codec);
        if (!m_decoder->initialize(config.decoderConfig)) {
            qCritical() << "Failed to initialize hardware decoder";
            return false;
        }
        
        // 3. 创建解封装器
        m_depacketizer = createDepacketizer(config.protocol);
        
        // 4. 启动管线线程
        m_decodeThread = std::make_unique<QThread>();
        m_decodeWorker = std::make_unique<DecodeWorker>(
            m_decoder.get(), m_framePool);
        m_decodeWorker->moveToThread(m_decodeThread.get());
        
        // 5. 连接信号槽（使用Qt::DirectConnection避免队列延迟）
        connect(m_decodeWorker.get(), &DecodeWorker::frameReady,
                this, &MediaPipeline::onFrameDecoded, 
                Qt::DirectConnection); // 关键：直连减少延迟
        
        m_decodeThread->start(QThread::TimeCriticalPriority);
        return true;
    }
    
    // 从网络接收视频包
    void onVideoPacketReceived(const uint8_t* data, size_t size, 
                                const PacketMetadata& meta) {
        // 在网络IO线程中调用
        auto nalUnits = m_depacketizer->depacketize(data, size, meta);
        
        for (auto& nalu : nalUnits) {
            // 送入解码队列（lock-free queue）
            m_decodeQueue.push(std::move(nalu));
            m_decodeWorker->notify(); // 唤醒解码线程
        }
    }

signals:
    void frameReady(std::shared_ptr<VideoFrame> frame);
    void pipelineError(PipelineError error);
    void statsUpdated(PipelineStats stats);
    
private:
    std::unique_ptr<IHardwareDecoder> m_decoder;
    std::shared_ptr<FramePool> m_framePool;
    std::unique_ptr<IDepacketizer> m_depacketizer;
    LockFreeQueue<NALUnit> m_decodeQueue;
    std::unique_ptr<QThread> m_decodeThread;
    std::unique_ptr<DecodeWorker> m_decodeWorker;
};

// =====================================================
// 帧池 - 预分配避免GC和分配延迟
// =====================================================
class FramePool {
public:
    FramePool(size_t poolSize, uint32_t maxW, uint32_t maxH, 
              VideoFrame::MemoryType memType) {
        for (size_t i = 0; i < poolSize; ++i) {
            auto frame = allocateGPUFrame(maxW, maxH, memType);
            m_freeFrames.push(std::move(frame));
        }
    }
    
    // 获取空闲帧（无锁，O(1)）
    std::shared_ptr<VideoFrame> acquire() {
        VideoFrame* frame = nullptr;
        if (m_freeFrames.pop(frame)) {
            return std::shared_ptr<VideoFrame>(frame, 
                [this](VideoFrame* f) { release(f); });
        }
        // 池耗尽时的策略：丢弃最老的帧 or 等待
        qWarning() << "Frame pool exhausted!";
        return nullptr;
    }
    
private:
    void release(VideoFrame* frame) {
        frame->reset(); // 重置元数据，保留GPU buffer
        m_freeFrames.push(frame);
    }
    
    LockFreeStack<VideoFrame*> m_freeFrames;
};

        m_samplerThread->start();
    }

// =====================================================
// 事件总线实现
// =====================================================
class EventBus : public QObject {
    Q_OBJECT
public:
    static EventBus& instance() {
        static EventBus bus;
        return bus;
    }
    
    // 订阅（类型安全）
    template<typename EventType>
    SubscriptionHandle subscribe(
            std::function<void(const EventType&)> handler,
            Qt::ConnectionType connType = Qt::AutoConnection) {
        
        uint32_t typeId = EventType::TYPE_ID;
        auto wrapper = [handler](const Event& e) {
            handler(static_cast<const EventType&>(e));
        };
        
        QMutexLocker lock(&m_mutex);
        auto handle = m_nextHandle++;
        m_subscribers[typeId].push_back({handle, wrapper, connType});
        return handle;
    }
    
    // 发布（按优先级分发）
    template<typename EventType>
    void publish(const EventType& event) {
        uint32_t typeId = EventType::TYPE_ID;
        
        if (event.priority() == Event::Priority::CRITICAL) {
            // 关键事件：同步直接调用，保证立即执行
            dispatchImmediate(typeId, event);
        } else {
            // 其他事件：进入优先级队列
            m_eventQueue.push(event.priority(), 
                              std::make_unique<EventType>(event));
            m_dispatchCondition.wakeOne();
        }
    }
    
    // 取消订阅
    void unsubscribe(SubscriptionHandle handle) {
        QMutexLocker lock(&m_mutex);
        for (auto& [typeId, subs] : m_subscribers) {
            subs.erase(
                std::remove_if(subs.begin(), subs.end(),
                    [handle](const auto& s) { return s.handle == handle; }),
                subs.end()
            );
        }
    }
    
private:
    void dispatchImmediate(uint32_t typeId, const Event& event) {
        QMutexLocker lock(&m_mutex);
        auto it = m_subscribers.find(typeId);
        if (it != m_subscribers.end()) {
            for (auto& sub : it->second) {
                sub.handler(event); // 直接调用
            }
        }
    }
    
    struct Subscriber {
        SubscriptionHandle handle;
        std::function<void(const Event&)> handler;
        Qt::ConnectionType connType;
    };
    
    PriorityQueue<std::unique_ptr<Event>> m_eventQueue;
    std::unordered_map<uint32_t, std::vector<Subscriber>> m_subscribers;
    QMutex m_mutex;
    QWaitCondition m_dispatchCondition;
    std::atomic<SubscriptionHandle> m_nextHandle{0};
};

// =====================================================
// 系统状态机 - 管理远程驾驶全生命周期

/*
 * 状态转换图：
 * 
 *  ┌──────────┐     connect     ┌──────────────┐
 *  │  IDLE    │───────────────▶│  CONNECTING   │
 *  │          │◀──timeout/fail──│              │
 *  └──────────┘                 └──────┬───────┘
 *                                      │ connected
 *  ┌──────────┐   auth_success  ┌──────▼───────┐
 *  │ READY    │◀────────────────│AUTHENTICATING│
 *  │          │                 └──────────────┘
 *  └────┬─────┘
 *       │ start_session
 *  ┌────▼─────────┐  preflight_ok  ┌──────────────┐
 *  │ PRE_FLIGHT   │──────────────▶│   DRIVING     │
 *  │ (系统自检)    │               │  (远程驾驶中)  │
 *  └──────────────┘               └───┬───────┬───┘
 *                                     │       │
 *                    network_degrade  │       │ emergency
 *                   ┌─────────────────┘       │
 *              ┌────▼──────────┐         ┌────▼──────────┐
 *              │  DEGRADED     │         │  EMERGENCY    │
 *              │ (降级运行)     │         │  (紧急停车)    │
 *              └───┬───────────┘         └───────────────┘
 *                  │ recovered                    │
 *                  └──▶ DRIVING                   │
 *                                                 │
 *  ┌──────────┐      stop_session          ┌─────▼────────┐
 *  │  IDLE    │◀───────────────────────────│  STOPPING    │
 *  └──────────┘                            └──────────────┘
 */

class SystemStateMachine : public QObject {
    Q_OBJECT
    Q_PROPERTY(SystemState currentState READ currentState 
               NOTIFY stateChanged)
public:
    enum class SystemState {
        IDLE,
        CONNECTING,
        AUTHENTICATING,
        READY,
        PRE_FLIGHT,
        DRIVING,
        DEGRADED,
        EMERGENCY,
        STOPPING,
        ERROR,
    };
    Q_ENUM(SystemState)
    
    // 状态转换触发器
    enum class Trigger {
        CONNECT,
        CONNECTED,
        AUTH_SUCCESS,
        AUTH_FAILURE,
        START_SESSION,
        PREFLIGHT_OK,
        PREFLIGHT_FAIL,
        EMERGENCY_STOP,
        NETWORK_DEGRADE,
        NETWORK_RECOVER,
        STOP_SESSION,
        CONNECTION_LOST,
        TIMEOUT,
        RESET,
    };

    SystemStateMachine(QObject* parent = nullptr) : QObject(parent) {
        setupTransitions();
    }
    
    bool fire(Trigger trigger) {
        QMutexLocker lock(&m_mutex);
        
        auto key = std::make_pair(m_currentState, trigger);
        auto it = m_transitions.find(key);
        
        if (it == m_transitions.end()) {
            qWarning() << "Invalid transition:" 
                       << stateToString(m_currentState) 
                       << "+" << triggerToString(trigger);
            return false;
        }
        
        auto& transition = it->second;
        
        // 检查守卫条件
        if (transition.guard && !transition.guard()) {
            qDebug() << "Transition guard rejected";
            return false;
        }
        
        SystemState oldState = m_currentState;
        SystemState newState = transition.targetState;
        
        // 执行退出动作
        if (m_exitActions.count(oldState)) {
            m_exitActions[oldState]();
        }
        
        // 执行转换动作
        if (transition.action) {
            transition.action();
        }
        
        // 状态切换
        m_currentState = newState;
        
        // 执行进入动作
        if (m_entryActions.count(newState)) {
            m_entryActions[newState]();
        }
        
        qInfo() << "State transition:" << stateToString(oldState) 
                << "->" << stateToString(newState);
        
        emit stateChanged(m_currentState, oldState);
        return true;
    }
    
    SystemState currentState() const { return m_currentState; }
    
signals:
    void stateChanged(SystemState newState, SystemState oldState);
    void emergencyActivated(const QString& reason);
    
private:
    struct Transition {
        SystemState targetState;
        std::function<bool()> guard;     // 守卫条件
        std::function<void()> action;    // 转换动作
    };
    
    void setupTransitions() {
        // IDLE状态的转换
        addTransition(SystemState::IDLE, Trigger::CONNECT, 
                      SystemState::CONNECTING, nullptr,
            [this]() { startConnectionSequence(); });
        
        // CONNECTING状态的转换
        addTransition(SystemState::CONNECTING, Trigger::CONNECTED,
                      SystemState::AUTHENTICATING, nullptr,
            [this]() { startAuthentication(); });
        
        addTransition(SystemState::CONNECTING, Trigger::TIMEOUT,
                      SystemState::IDLE, nullptr,
            [this]() { handleConnectionTimeout(); });
        
        // DRIVING状态的转换
        addTransition(SystemState::DRIVING, Trigger::EMERGENCY_STOP,
                      SystemState::EMERGENCY, nullptr,
            [this]() { 
                executeEmergencyStop(); 
                emit emergencyActivated("Emergency stop triggered");
            });
        
        addTransition(SystemState::DRIVING, Trigger::NETWORK_DEGRADE,
                      SystemState::DEGRADED,
            [this]() -> bool { 
                // 只有在网络质量低于阈值时才降级
                return m_networkQuality.score < 0.5; 
            },
            [this]() { activateDegradedMode(); });
        
        // DEGRADED的恢复
        addTransition(SystemState::DEGRADED, Trigger::NETWORK_RECOVER,
                      SystemState::DRIVING,
            [this]() -> bool {
                // 网络质量需持续恢复超过2秒
                return m_networkRecoveryDuration > 2000;
            },
            [this]() { deactivateDegradedMode(); });
        
        // 任何驾驶状态都可以紧急停车
        addTransition(SystemState::DEGRADED, Trigger::EMERGENCY_STOP,
                      SystemState::EMERGENCY);
        
        // ... 更多转换规则
        
        // 设置状态进入/退出动作
        m_entryActions[SystemState::DRIVING] = [this]() {
            startControlLoop();
            startTelemetryMonitoring();
            startSafetyWatchdog();
        };
        
        m_exitActions[SystemState::DRIVING] = [this]() {
            stopControlLoop();
        };
        
        m_entryActions[SystemState::EMERGENCY] = [this]() {
            sendEmergencyStopToVehicle();
            disableAllControls();
            startEmergencyRecording();
        };
    }
    
    void addTransition(SystemState from, Trigger trigger, SystemState to,
                       std::function<bool()> guard = nullptr,
                       std::function<void()> action = nullptr) {
        m_transitions[{from, trigger}] = {to, guard, action};
    }
    
    std::map<std::pair<SystemState, Trigger>, Transition> m_transitions;
    std::map<SystemState, std::function<void()>> m_entryActions;
    std::map<SystemState, std::function<void()>> m_exitActions;
    
    SystemState m_currentState = SystemState::IDLE;
    QMutex m_mutex;
    NetworkQuality m_networkQuality;
    int64_t m_networkRecoveryDuration = 0;
};

class PluginManager : public QObject {
    Q_OBJECT
public:
    void scanPlugins(const QString& pluginDir) {
        QDir dir(pluginDir);
        for (const auto& file : dir.entryList({"*.so", "*.dll", "*.dylib"})) {
            auto loader = std::make_unique<QPluginLoader>(
                dir.absoluteFilePath(file));
            
            if (auto plugin = qobject_cast<IPlugin*>(loader->instance())) {
                auto info = plugin->info();
                m_plugins[info.id] = {std::move(loader), plugin, info};
                qInfo() << "Discovered plugin:" << info.name.c_str();
            }
        }
    }
    
    bool loadPlugin(const std::string& pluginId) {
        auto it = m_plugins.find(pluginId);
        if (it == m_plugins.end()) return false;
        
        // 检查依赖
        for (const auto& dep : it->second.plugin->dependencies()) {
            if (!isPluginLoaded(dep)) {
                if (!loadPlugin(dep)) {
                    qCritical() << "Failed to load dependency:" << dep.c_str();
                    return false;
                }
            }
        }
        
        return it->second.plugin->initialize(m_context);
    }
    
    // 按优先级排序后批量加载
    void loadAllPlugins() {
        std::vector<std::pair<std::string, PluginEntry*>> sorted;
        for (auto& [id, entry] : m_plugins) {
            sorted.push_back({id, &entry});
        }
        std::sort(sorted.begin(), sorted.end(),
            [](const auto& a, const auto& b) {
                return a.second->plugin->priority() > b.second->plugin->priority();
            });
        
        for (auto& [id, entry] : sorted) {
            loadPlugin(id);
        }
    }
    
private:
    struct PluginEntry {
        std::unique_ptr<QPluginLoader> loader;
        IPlugin* plugin;
        PluginInfo info;
    };
    
    std::map<std::string, PluginEntry> m_plugins;
    PluginContext m_context;
};

class VehicleControlService : public QObject {
    Q_OBJECT
public:
    /*
     * 控制环路设计：
     * 
     * ┌──────────┐    ┌───────────┐    ┌──────────┐    ┌──────────┐
     * │  Input   │──▶│  Control  │──▶│ Safety   │──▶│ Network  │
     * │ Sampler  │   │ Processor │   │ Checker  │   │ Sender   │
     * │ (200Hz)  │   │ (100Hz)   │   │          │   │ (100Hz)  │
     * └──────────┘   └───────────┘   └──────────┘   └──────────┘
     *                     │
     *                ┌────▼─────┐
     *                │Predictor │ (延迟补偿)
     *                └──────────┘
     */
    
    struct ControlCommand {
        double steeringAngle;     // rad
        double throttle;          // [0, 1]
        double brake;             // [0, 1]
        int gear;
        bool emergencyStop;
        int64_t timestamp;        // 发送时间戳
        uint32_t sequenceNumber;
        
        // 延迟补偿预测值
        struct Prediction {
            double predictedSteeringAngle;
            double predictionHorizonMs;
        } prediction;
    };
    
    bool initialize(const ControlConfig& config) {
        m_config = config;
        
        // 创建控制环定时器
        m_controlTimer = std::make_unique<QTimer>();
        m_controlTimer->setTimerType(Qt::PreciseTimer);
        m_controlTimer->setInterval(1000 / config.controlRateHz); // 10ms = 100Hz
        
        connect(m_controlTimer.get(), &QTimer::timeout, 
                this, &VehicleControlService::controlTick);
        
        // 初始化延迟补偿预测器
        m_predictor = std::make_unique<LatencyCompensator>(config.prediction);
        
        // 初始化安全检查器
        m_safetyChecker = std::make_unique<SafetyChecker>(config.safety);
        
        return true;
    }
    
    void start() {
        m_controlTimer->start();
        m_isRunning = true;
        qInfo() << "Control loop started at" << m_config.controlRateHz << "Hz";
    }
    
    void stop() {
        m_controlTimer->stop();
        m_isRunning = false;
        // 发送中性控制指令
        sendNeutralCommand();
    }
    
    // 更新输入状态（从InputSampler调用）
    void updateInput(const IInputDevice::InputState& input) {
        // 原子更新，无锁
        m_latestInput.store(input);
    }
    
private slots:
    void controlTick() {
        auto tickStart = highResolutionClock::now();
        
        // 1. 读取最新输入
        auto input = m_latestInput.load();
        
        // 2. 输入处理（非线性映射、曲线、死区）
        auto processed = processInput(input);
        
        // 3. 延迟补偿预测
        auto currentRTT = m_networkMonitor->getCurrentRTT();
        auto prediction = m_predictor->predict(processed, currentRTT);
        
        // 4. 速率限制（防止过度转向）
        auto rateLimited = applyRateLimit(prediction, m_lastCommand);
        
        // 5. 安全检查
        auto safetyResult = m_safetyChecker->check(rateLimited, 
                                                     m_vehicleState);
        if (!safetyResult.safe) {
            handleSafetyViolation(safetyResult);
            return;
        }
        
        // 6. 构造控制命令
        ControlCommand cmd;
        cmd.steeringAngle = rateLimited.steeringAngle;
        cmd.throttle = rateLimited.throttle;
        cmd.brake = rateLimited.brake;
        cmd.gear = rateLimited.gear;
        cmd.emergencyStop = false;
        cmd.timestamp = highResolutionTimestamp();
        cmd.sequenceNumber = m_sequenceCounter++;
        cmd.prediction = prediction;
        
        // 7. 发送（通过高优先级通道）
        sendCommand(cmd);
        
        m_lastCommand = cmd;
        
        // 8. 性能监控
        auto tickDuration = highResolutionClock::now() - tickStart;
        m_controlLatencyStats.addSample(
            std::chrono::duration_cast<std::chrono::microseconds>(tickDuration).count()
        );
        
        if (m_controlLatencyStats.p99() > 5000) { // > 5ms
            qWarning() << "Control tick P99 latency too high:" 
                       << m_controlLatencyStats.p99() << "us";
        }
    }
    
private:
    // 输入非线性映射
    ProcessedInput processInput(const IInputDevice::InputState& raw) {
        ProcessedInput result;
        
        // 方向盘：应用非线性曲线（低速精确，高速灵敏）
        result.steeringAngle = applyCurve(raw.steeringAngle, 
                                           m_config.steeringCurve);
        
        // 油门：应用死区和渐进曲线
        result.throttle = applyDeadzone(raw.throttle, 
                                         m_config.throttleDeadzone);
        result.throttle = applyCurve(result.throttle, 
                                      m_config.throttleCurve);
        
        // 刹车：线性但有死区
        result.brake = applyDeadzone(raw.brake, m_config.brakeDeadzone);
        
        return result;
    }
    
    // 速率限制（安全措施）
    ProcessedInput applyRateLimit(const ProcessedInput& desired,
                                   const ControlCommand& last) {
        ProcessedInput limited = desired;
        
        double dt = 1.0 / m_config.controlRateHz;
        double maxSteeringRate = m_config.maxSteeringRateRadPerSec * dt;
        
        double steeringDelta = desired.steeringAngle - last.steeringAngle;
        if (std::abs(steeringDelta) > maxSteeringRate) {
            limited.steeringAngle = last.steeringAngle + 
                std::copysign(maxSteeringRate, steeringDelta);
        }
        
        return limited;
    }
    
    void sendCommand(const ControlCommand& cmd) {
        // 序列化
        auto packet = serialize(cmd);
        
        // 通过控制通道发送（最高优先级UDP通道）
        m_transport->send(TransportChannel::CONTROL_CRITICAL,
                          packet.data(), packet.size(),
                          SendFlags::Unreliable | SendFlags::NoCopy);
    }
    
    // 安全看门狗
    void handleSafetyViolation(const SafetyCheckResult& result) {
        qCritical() << "Safety violation:" << result.reason;
        
        switch (result.severity) {
            case SafetySeverity::WARNING:
                emit safetyWarning(result.reason);
                break;
            case SafetySeverity::LIMIT:
                // 限制控制输出
                sendLimitedCommand(result.limitedCommand);
                break;
            case SafetySeverity::CRITICAL:
                // 触发紧急停车
                EventBus::instance().publish(EmergencyStopEvent{
                    .reason = result.reason,
                    .source = EmergencyStopSource::SAFETY_CHECKER
                });
                break;
        }
    }
    
    std::atomic<IInputDevice::InputState> m_latestInput; // 无锁输入
    std::unique_ptr<QTimer> m_controlTimer;
    std::unique_ptr<LatencyCompensator> m_predictor;
    std::unique_ptr<SafetyChecker> m_safetyChecker;
    ControlCommand m_lastCommand;
    ControlConfig m_config;
    std::atomic<uint32_t> m_sequenceCounter{0};
    PercentileStats m_controlLatencyStats;
    bool m_isRunning = false;
};

// =====================================================
// 延迟补偿预测器
// =====================================================
class LatencyCompensator {
public:
    struct PredictionResult {
        double predictedSteeringAngle;
        double predictionHorizonMs;
        double confidence;
    };
    
    PredictionResult predict(const ProcessedInput& current, 
                              double currentRTTMs) {
        // 预测范围 = RTT/2 + 处理延迟
        double predictionHorizon = currentRTTMs / 2.0 + m_processingDelayMs;
        
        // 基于历史输入的趋势预测
        m_inputHistory.push_back({highResolutionTimestamp(), current});
        
        if (m_inputHistory.size() < 3) {
            return {current.steeringAngle, predictionHorizon, 0.5};
        }
        
        // 使用二次多项式拟合预测
        double predicted = quadraticExtrapolation(
            m_inputHistory, predictionHorizon);
        
        // 限制预测范围
        double maxPredictionDelta = m_maxPredictionAngle;
        predicted = std::clamp(predicted,
            current.steeringAngle - maxPredictionDelta,
            current.steeringAngle + maxPredictionDelta);
        
        return {predicted, predictionHorizon, calculateConfidence()};
    }
    
private:
    CircularBuffer<std::pair<int64_t, ProcessedInput>, 50> m_inputHistory;
    double m_processingDelayMs = 5.0;
    double m_maxPredictionAngle = 0.1; // rad
};

        status.latencyOk = latencyCheck.withinLimits;

// =====================================================
// 线程间通信：根据场景选择最优方式
// =====================================================

class ThreadCommunicationStrategy {
public:
    /*
     * 通信方式选择矩阵：
     * 
     * ┌──────────────────┬────────────────┬──────────────┬─────────────┐
     * │ 场景              │ 方式           │ 延迟        │ 线程安全    │
     * ├──────────────────┼────────────────┼──────────────┼─────────────┤
     * │ 输入→控制         │ atomic变量     │ < 100ns     │ lock-free   │
     * │ 网络→解码         │ SPSC无锁队列   │ < 500ns     │ lock-free   │
     * │ 解码→渲染         │ 三缓冲+原子    │ < 1us       │ lock-free   │
     * │ 控制→网络         │ SPSC无锁队列   │ < 500ns     │ lock-free   │
     * │ 安全→状态机       │ Qt::Direct     │ < 1us       │ signal/slot │
     * │ 业务→UI           │ Qt::Queued     │ < 1ms       │ signal/slot │
     * │ 诊断→存储         │ MPSC队列       │ 不关心      │ lock-free   │
     * └──────────────────┴────────────────┴──────────────┴─────────────┘
     */
    
    // SPSC无锁队列（单生产者单消费者）
    template<typename T, size_t Capacity = 1024>
    class SPSCQueue {
    public:
        bool push(const T& item) {
            auto current_tail = m_tail.load(std::memory_order_relaxed);
            auto next_tail = (current_tail + 1) % Capacity;
            if (next_tail == m_head.load(std::memory_order_acquire)) {
                return false; // 队列满
            }
            m_buffer[current_tail] = item;
            m_tail.store(next_tail, std::memory_order_release);
            return true;
        }
        
        bool pop(T& item) {
            auto current_head = m_head.load(std::memory_order_relaxed);
            if (current_head == m_tail.load(std::memory_order_acquire)) {
                return false; // 队列空
            }
            item = m_buffer[current_head];
            m_head.store((current_head + 1) % Capacity, 
                         std::memory_order_release);
            return true;
        }
        
    private:
        alignas(64) std::array<T, Capacity> m_buffer;
        alignas(64) std::atomic<size_t> m_head{0};
        alignas(64) std::atomic<size_t> m_tail{0};
    };
    
    // 三缓冲（Triple Buffering）- 用于视频帧传递
    template<typename T>
    class TripleBuffer {
    public:
        // 生产者写入
        T& getWriteBuffer() {
            return m_buffers[m_writeIndex];
        }
        
        void publishWrite() {
            // 原子交换write和middle
            auto oldMiddle = m_middleIndex.exchange(
                m_writeIndex, std::memory_order_acq_rel);
            m_writeIndex = oldMiddle;
            m_newData.store(true, std::memory_order_release);
        }
        
        // 消费者读取
        bool hasNewData() const {
            return m_newData.load(std::memory_order_acquire);
        }
        
        const T& getReadBuffer() {
            if (m_newData.exchange(false, std::memory_order_acq_rel)) {
                // 原子交换read和middle
                auto oldMiddle = m_middleIndex.exchange(
                    m_readIndex, std::memory_order_acq_rel);
                m_readIndex = oldMiddle;
            }
            return m_buffers[m_readIndex];
        }
        
    private:
        std::array<T, 3> m_buffers;
        int m_writeIndex = 0;
        int m_readIndex = 2;
        std::atomic<int> m_middleIndex{1};
        std::atomic<bool> m_newData{false};
    };
};

// =====================================================

class MemoryManager {
public:
    /*
     * 策略：
     * 1. 预分配所有关键路径的内存
     * 2. 使用对象池避免运行时分配
     * 3. 固定大小缓冲区避免碎片化
     * 4. 关键路径上禁止std::string等动态分配
     */
    
    // 启动时预分配
    void preallocate() {
        // 视频帧池 (GPU memory)
        m_videoFramePool.preallocate(16, MAX_FRAME_SIZE);
        
        // 网络包池
        m_networkPacketPool.preallocate(256, MAX_PACKET_SIZE);
        
        // 控制命令池
        m_controlCommandPool.preallocate(32, sizeof(ControlCommand));
        
        // 锁定关键内存页，防止被换出
        #ifdef Q_OS_LINUX
        mlockall(MCL_CURRENT | MCL_FUTURE);
        #endif
    }
    
    // 对象池模板
    template<typename T, size_t PoolSize>
    class ObjectPool {
    public:
        void preallocate(size_t count, size_t objectSize) {
            m_storage.resize(count * objectSize);
            for (size_t i = 0; i < count; ++i) {
                m_freeList.push(reinterpret_cast<T*>(
                    m_storage.data() + i * objectSize));
            }
        }
        
        T* acquire() {
            T* obj = nullptr;
            if (m_freeList.pop(obj)) {
                return obj;
            }
            qWarning() << "Object pool exhausted for" << typeid(T).name();
            return nullptr; // 不fallback到堆分配
        }
        
        void release(T* obj) {
            obj->~T(); // 调用析构
            m_freeList.push(obj);
        }
        
    private:
        std::vector<uint8_t> m_storage;
        LockFreeStack<T*> m_freeList;
    };
    
private:
    ObjectPool<VideoFrame, 16> m_videoFramePool;
    ObjectPool<NetworkPacket, 256> m_networkPacketPool;
    ObjectPool<ControlCommand, 32> m_controlCommandPool;

// =====================================================
// 自适应降级管理器
// =====================================================

class DegradationManager : public QObject {
    Q_OBJECT
public:
    enum class DegradationLevel {
        FULL_QUALITY,        // 全质量：4K 60fps 多路视频
        HIGH_QUALITY,        // 高质量：1080p 30fps 多路
        MEDIUM_QUALITY,      // 中质量：720p 30fps 主视角+缩略图
        LOW_QUALITY,         // 低质量：480p 15fps 仅主视角
        MINIMAL,             // 最低：360p 10fps 仅主视角 + 限速
        SAFETY_STOP,         // 安全停车：停止一切远程驾驶
    };
    
    struct DegradationProfile {
        DegradationLevel level;
        
        struct VideoConfig {
            int mainCameraWidth, mainCameraHeight;
            int mainCameraFps;
            bool enableSecondaryCameras;
            int secondaryCameraFps;
            int targetBitrateKbps;
        } video;
        
        struct ControlConfig {
            int controlRateHz;
            double maxSpeedKmh;
            bool enablePrediction;
        } control;
        
        struct UIConfig {
            bool showMiniMap;
            bool showTelemetryDetails;
            int hudUpdateRateHz;
        } ui;
    };
    
    void evaluateAndApply(const NetworkQuality& quality) {
        DegradationLevel targetLevel = determineLevel(quality);
        
        if (targetLevel != m_currentLevel) {
            // 防止频繁切换（滞后机制）
            if (shouldTransition(targetLevel)) {
                applyDegradation(targetLevel);
                m_currentLevel = targetLevel;
                emit degradationLevelChanged(m_currentLevel);
            }
        }
    }
    
private:
    DegradationLevel determineLevel(const NetworkQuality& q) {
        // 基于多维度评分
        if (q.rttMs < 50 && q.packetLoss < 0.01 && q.bandwidthKbps > 20000)
            return DegradationLevel::FULL_QUALITY;
        if (q.rttMs < 100 && q.packetLoss < 0.03 && q.bandwidthKbps > 10000)
            return DegradationLevel::HIGH_QUALITY;
        if (q.rttMs < 150 && q.packetLoss < 0.05 && q.bandwidthKbps > 5000)
            return DegradationLevel::MEDIUM_QUALITY;
        if (q.rttMs < 200 && q.packetLoss < 0.10 && q.bandwidthKbps > 2000)
            return DegradationLevel::LOW_QUALITY;
        if (q.rttMs < 300 && q.packetLoss < 0.20)
            return DegradationLevel::MINIMAL;
            
        return DegradationLevel::SAFETY_STOP;
    }
    
    bool shouldTransition(DegradationLevel target) {
        auto now = QDateTime::currentMSecsSinceEpoch();
        
        // 降级快（立即），升级慢（需持续稳定3秒）
        if (target > m_currentLevel) { // 降级
            return true;
        } else { // 升级
            if (m_upgradeCandidate != target) {
                m_upgradeCandidate = target;
                m_upgradeCandidateTime = now;
                return false;
            }
            return (now - m_upgradeCandidateTime) > 3000; // 3秒稳定
        }
    }
    
    void applyDegradation(DegradationLevel level) {
        auto profile = getProfile(level);
        
        // 应用视频配置
        m_mediaPipeline->reconfigure(profile.video);
        
        // 应用控制配置
        m_controlService->setControlRate(profile.control.controlRateHz);
        m_controlService->setMaxSpeed(profile.control.maxSpeedKmh);
        
        // 应用UI配置
        emit uiConfigChanged(profile.ui);
        
        qInfo() << "Degradation level changed to:" << static_cast<int>(level);
    }
    
    DegradationLevel m_currentLevel = DegradationLevel::FULL_QUALITY;
    DegradationLevel m_upgradeCandidate;
    int64_t m_upgradeCandidateTime = 0;
};

// =====================================================
// 分级错误恢复策略
// =====================================================

class ErrorRecoveryManager : public QObject {
    Q_OBJECT
public:
    /*
     * 错误恢复等级：
     * 
     * Level 1 - 自动重试 (网络抖动、单帧丢失)
     *   → 自动重连、请求关键帧、插值补帧
     * 
     * Level 2 - 服务重启 (解码器错误、通道断开)
     *   → 重启特定服务模块，不中断整体会话
     * 
     * Level 3 - 会话重建 (严重网络中断 > 3秒)
     *   → 重新建立完整会话，车辆自动减速等待
     * 
     * Level 4 - 安全停车 (不可恢复错误)
     *   → 触发车辆自主安全停车，通知监控中心
     */
    
    void handleError(const SystemError& error) {
        auto level = classifyError(error);
        auto& tracker = m_errorTrackers[error.module];
        
        tracker.recordError(error);
        
        // 检查是否需要升级恢复等级
        if (tracker.errorRatePerMinute() > m_escalationThreshold) {
            level = escalateLevel(level);
        }
        
        switch (level) {
            case RecoveryLevel::AUTO_RETRY:
                handleAutoRetry(error);
                break;
            case RecoveryLevel::SERVICE_RESTART:
                handleServiceRestart(error);
                break;
            case RecoveryLevel::SESSION_REBUILD:
                handleSessionRebuild(error);
                break;
            case RecoveryLevel::SAFETY_STOP:
                handleSafetyStop(error);
                break;
        }
    }
    
private:
    void handleAutoRetry(const SystemError& error) {
        switch (error.type) {
            case ErrorType::VIDEO_FRAME_LOSS:
                // 请求IDR帧
                m_transport->send(TransportChannel::SIGNALING,
                    buildIDRRequest());
                break;
                
            case ErrorType::NETWORK_PACKET_LOSS:
                // FEC已经在处理，可能需要请求NACK
                m_transport->sendNACK(error.sequenceNumbers);
                break;
                
            case ErrorType::DECODE_ERROR:
                // 刷新解码器
                m_decoder->flush();
                // 请求新的关键帧
                requestKeyFrame();
                break;
        }
    }
    
    void handleServiceRestart(const SystemError& error) {
        qWarning() << "Restarting service:" << error.module;
        
        auto service = m_serviceRegistry->get(error.module);
        if (service) {
            service->shutdown();
            QThread::msleep(100);
            service->initialize(m_context);
            
            emit serviceRestarted(error.module);
        }
    }
    
    void handleSessionRebuild(const SystemError& error) {
        qCritical() << "Rebuilding session due to:" << error.message;
        
        // 1. 通知车辆自动减速
        sendVehicleSlowDown();
        
        // 2. 重建所有通道
        m_sessionManager->rebuild();
        
        // 3. 恢复后验证
        if (!verifySession()) {
            handleSafetyStop(error);
        }
    }
    
    void handleSafetyStop(const SystemError& error) {
        qCritical() << "SAFETY STOP:" << error.message;
        
        EventBus::instance().publish(EmergencyStopEvent{
            .reason = "Unrecoverable error: " + error.message,
            .source = EmergencyStopSource::ERROR_RECOVERY
        });
    }
    
    std::map<std::string, ErrorTracker> m_errorTrackers;
    int m_escalationThreshold = 10; // 每分钟10次错误升级
};

class PerformanceMonitor : public QObject {

            double cpuUsagePercent;
            double gpuUsagePercent;

        } resources;

        metrics.throughput.controlHz = m_controlRateCounter.currentHz();

                       << metrics.latency.uiFrameTimeMs << "ms";
        }

            qWarning() << "Video E2E latency exceeds 100ms:"
                       << metrics.latency.videoE2EMs << "ms";

        // 上报遥测
        emit metricsUpdated(metrics);
        
        // 本地存储（环形缓冲，用于事后分析）
        m_metricsHistory.push(metrics);
    }

    public:
        ScopedTimer(PercentileTracker& tracker) 
            : m_tracker(tracker)
            , m_start(std::chrono::high_resolution_clock::now()) {}
        
        ~ScopedTimer() {
            auto elapsed = std::chrono::high_resolution_clock::now() - m_start;
            auto us = std::chrono::duration_cast<

            m_tracker.addSample(us);

    private:

        std::chrono::high_resolution_clock::time_point m_start;
    };
    
signals:

    void performanceAlert(const QString& message, AlertSeverity severity);
    
private:

    FPSCounter m_fpsCounter;
    RateCounter m_controlRateCounter;
